```139:151:image_analysis_lib/musiq.py
def _musiq_score_for_csv(score: Optional[float]) -> str:
    """Format score for MUSIQ CSV: three decimal places; values >= 10 become 9.999."""
    if score is None:
        return ""
    x = float(score)
    if x >= 10.0:
        x = 9.999
    elif x < 0.0:
        x = 0.0
    return f"{x:.3f}"
```
